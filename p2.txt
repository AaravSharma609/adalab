#include <stdio.h>
#include <stdlib.h>
#include <time.h>

void swap(int *a, int *b)
{
    int temp = *a;
    *a = *b;
    *b = temp;
}

int partition(int arr[], int low, int high)
{
    int pivot = arr[low]; 
    int i = low + 1;
    int j = high;

    while (1)
    {
        while (i <= high && arr[i] <= pivot)
            i++;

        while (arr[j] > pivot)
            j--;

        if (i >= j)
            break;

        swap(&arr[i], &arr[j]);
    }

    swap(&arr[low], &arr[j]);
    return j;
}

void quickSort(int arr[], int low, int high)
{
    if (low < high)
    {
        int pi = partition(arr, low, high);
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
}

int main()
{
    FILE *fp;
    fp = fopen("time_data.dat", "w");

    if (fp == NULL)
    {
        printf("Error opening file!\n");
        return 1;
    }

    srand(time(NULL));

    printf("Generating time data...\n");

    for (int n = 5000; n <= 50000; n += 5000)
    {
        int arr[n];

        for (int i = 0; i < n; i++)
            arr[i] = rand() % 1000;

        clock_t start, end;

        start = clock();
        quickSort(arr, 0, n - 1);
        end = clock();

        double time_taken = ((double)(end - start)) / CLOCKS_PER_SEC;

        printf("n = %d, time = %f seconds\n", n, time_taken);

        fprintf(fp, "%d %f\n", n, time_taken);
    }

    fclose(fp);

    printf("Time data saved to time_data.dat\n");

    FILE *gp = fopen("plot_script.gp", "w");

    fprintf(gp, "set title 'Quick Sort Time Complexity'\n");
    fprintf(gp, "set xlabel 'Number of Elements (n)'\n");
    fprintf(gp, "set ylabel 'Time (seconds)'\n");
    fprintf(gp, "set grid\n");
    fprintf(gp, "set terminal png\n");
    fprintf(gp, "set output 'quicksort_time.png'\n");
    fprintf(gp, "plot 'time_data.dat' using 1:2 with linespoints title 'QuickSort'\n");

    fclose(gp);

    system("gnuplot plot_script.gp");

    printf("Graph generated: quicksort_time.png\n");

    return 0;
}
